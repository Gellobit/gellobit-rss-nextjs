import React, { useState } from 'react';
import { 
  Save, 
  Image as ImageIcon, 
  Type, 
  Hash, 
  Layout, 
  Eye, 
  Globe, 
  Settings, 
  ChevronLeft,
  UploadCloud,
  CheckCircle,
  AlertCircle
} from 'lucide-react';

// Nota: En una app real usaríamos 'react-quill' o 'tiptap' para el área de texto.
// Aquí simulamos la interfaz del CMS personalizado.

const App = () => {
  const [postData, setPostData] = useState({
    title: '',
    excerpt: '',
    category: 'Cheap and Near',
    content: '',
    isEvergreen: true,
    status: 'draft'
  });

  const [isPublishing, setIsPublishing] = useState(false);
  const [showSuccess, setShowSuccess] = useState(false);

  const handlePublish = () => {
    setIsPublishing(true);
    // Simulación de guardado en Supabase
    setTimeout(() => {
      setIsPublishing(false);
      setShowSuccess(true);
      setTimeout(() => setShowSuccess(false), 3000);
    }, 1500);
  };

  return (
    <div className="min-h-screen bg-slate-50 flex font-sans text-slate-900">
      
      {/* SIDEBAR DE CONTROL */}
      <aside className="w-80 bg-white border-r border-slate-200 p-6 flex flex-col gap-8 hidden lg:flex">
        <div className="flex items-center gap-3">
          <div className="bg-yellow-400 p-2 rounded-lg font-bold">GB</div>
          <span className="font-black text-xl">Admin CMS</span>
        </div>

        <div className="space-y-6">
          <div>
            <label className="text-xs font-black uppercase text-slate-400 tracking-widest block mb-3">Configuración del Post</label>
            <div className="space-y-4">
              <div>
                <label className="text-sm font-bold block mb-2">Categoría</label>
                <select 
                  className="w-full p-3 bg-slate-50 border border-slate-200 rounded-xl outline-none focus:ring-2 focus:ring-yellow-400"
                  value={postData.category}
                  onChange={(e) => setPostData({...postData, category: e.target.value})}
                >
                  <option>Giveaways</option>
                  <option>Job Fairs</option>
                  <option>Scholarships</option>
                  <option>Cheap and Near</option>
                </select>
              </div>

              <div className="flex items-center justify-between p-3 bg-slate-50 rounded-xl border border-slate-100">
                <span className="text-sm font-bold">Contenido Evergreen</span>
                <input 
                  type="checkbox" 
                  checked={postData.isEvergreen}
                  onChange={(e) => setPostData({...postData, isEvergreen: e.target.checked})}
                  className="w-5 h-5 accent-yellow-400"
                />
              </div>
            </div>
          </div>

          <div>
            <label className="text-xs font-black uppercase text-slate-400 tracking-widest block mb-3">SEO & Indexing</label>
            <div className="p-4 bg-green-50 rounded-2xl border border-green-100">
              <div className="flex items-center gap-2 text-green-700 font-bold text-xs mb-1">
                <Globe size={14} /> Google Indexing API
              </div>
              <p className="text-[10px] text-green-600 font-medium">Auto-notificar al publicar activo.</p>
            </div>
          </div>
        </div>

        <div className="mt-auto">
          <button 
            onClick={handlePublish}
            disabled={isPublishing}
            className="w-full bg-slate-900 text-white font-bold py-4 rounded-2xl flex items-center justify-center gap-2 hover:bg-slate-800 transition-all shadow-xl shadow-slate-200 disabled:opacity-50"
          >
            {isPublishing ? 'Publicando...' : <><Save size={18} /> Publicar Post</>}
          </button>
        </div>
      </aside>

      {/* EDITOR PRINCIPAL */}
      <main className="flex-1 flex flex-col h-screen overflow-hidden">
        
        {/* TOP BAR EDITOR */}
        <header className="h-20 bg-white border-b border-slate-200 px-8 flex items-center justify-between">
          <div className="flex items-center gap-4 text-slate-400">
            <ChevronLeft className="cursor-pointer hover:text-slate-900" />
            <span className="text-sm font-medium">Posts / Nuevo Artículo</span>
          </div>
          <div className="flex items-center gap-3">
            <button className="flex items-center gap-2 px-4 py-2 text-sm font-bold text-slate-600 hover:bg-slate-100 rounded-xl transition-all">
              <Eye size={18} /> Previsualizar
            </button>
          </div>
        </header>

        {/* ÁREA DE TRABAJO */}
        <div className="flex-1 overflow-y-auto p-8 md:p-12">
          <div className="max-w-4xl mx-auto space-y-8">
            
            {/* ÉXITO AL PUBLICAR */}
            {showSuccess && (
              <div className="bg-emerald-500 text-white p-4 rounded-2xl flex items-center gap-3 animate-bounce">
                <CheckCircle size={20} />
                <span className="font-bold text-sm">¡Post publicado y enviado a Google con éxito!</span>
              </div>
            )}

            {/* TÍTULO */}
            <input 
              type="text" 
              placeholder="Título del Artículo (ej: Dónde encontrar neumáticos gratis en USA)"
              className="text-4xl md:text-5xl font-black w-full bg-transparent border-none outline-none placeholder:text-slate-200 focus:ring-0"
              value={postData.title}
              onChange={(e) => setPostData({...postData, title: e.target.value})}
            />

            {/* EXCERPT / RESUMEN */}
            <div className="p-6 bg-yellow-50 rounded-3xl border border-yellow-100">
              <label className="text-[10px] font-black uppercase text-yellow-600 tracking-widest block mb-2">Resumen (SEO Excerpt)</label>
              <textarea 
                placeholder="Escribe un breve resumen para los resultados de búsqueda..."
                className="w-full bg-transparent border-none outline-none text-slate-700 font-medium resize-none text-lg h-20"
                value={postData.excerpt}
                onChange={(e) => setPostData({...postData, excerpt: e.target.value})}
              />
            </div>

            {/* IMAGEN DE PORTADA MOCK */}
            <div className="aspect-video w-full bg-slate-100 rounded-[40px] border-2 border-dashed border-slate-200 flex flex-col items-center justify-center gap-4 group hover:bg-slate-200 transition-all cursor-pointer">
              <div className="bg-white p-4 rounded-full shadow-sm group-hover:scale-110 transition-transform">
                <UploadCloud size={32} className="text-slate-400" />
              </div>
              <div className="text-center">
                <p className="font-bold text-slate-500">Subir Imagen de Portada</p>
                <p className="text-xs text-slate-400 mt-1">Recomendado: 1200 x 630px (WebP)</p>
              </div>
            </div>

            {/* CONTENIDO (SIMULACIÓN WYSIWYG) */}
            <div className="bg-white rounded-[40px] border border-slate-200 shadow-sm overflow-hidden">
              <div className="bg-slate-50 px-6 py-4 border-b border-slate-200 flex flex-wrap gap-4">
                 <button className="p-2 hover:bg-white rounded-lg text-slate-600"><Type size={18}/></button>
                 <button className="p-2 hover:bg-white rounded-lg text-slate-600 font-bold">B</button>
                 <button className="p-2 hover:bg-white rounded-lg text-slate-600 italic">I</button>
                 <button className="p-2 hover:bg-white rounded-lg text-slate-600"><ImageIcon size={18}/></button>
                 <button className="p-2 hover:bg-white rounded-lg text-slate-600 font-bold">H1</button>
                 <button className="p-2 hover:bg-white rounded-lg text-slate-600 font-bold">H2</button>
              </div>
              <textarea 
                placeholder="Empieza a escribir tu historia aquí..."
                className="w-full p-8 min-h-[500px] border-none outline-none text-lg leading-relaxed text-slate-700 resize-none"
                value={postData.content}
                onChange={(e) => setPostData({...postData, content: e.target.value})}
              />
            </div>

          </div>
        </div>
      </main>
    </div>
  );
};

export default App;
